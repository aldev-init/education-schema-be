//Query untuk menampilkan peserta dengan gaji tertinggi
//menampilkan kolom nama dan gaji tertinggi 
-select nama,MAX(gaji) as gaji from kawahedukasi;

//Query untuk menampilkan peserta Backend dengan nilai dan gaji tertinggi
//menampilkan kolom nama,nilai,gaji 
-select nama,MAX(nilai) as nilai,MAX(gaji) as gaji from kawahedukasi where kelas='BACKEND'

//Query untuk menampilkan total peserta yang berada dalam kelas frontend 
-select COUNT(*) as total_peserta_frontend from kawahedukasi WHERE kelas='FRONTEND'

//Query untuk menampilkan jumlah kehadiran tertinggi dan nilai tertinggi dalam kelas frontend
-select nama,MAX(jumlah_kehadiran) as jumlah_kehadiran,MAX(nilai) as nilai from kawahedukasi where kelas='FRONTEND'
